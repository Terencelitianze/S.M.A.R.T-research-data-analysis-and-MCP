def plot_data(df, column, title, ylabel):
    """
    Plots the data for a specific column over time with improved date formatting.
    
    Parameters:
    df (pd.DataFrame): The dataframe containing the data to plot.
    column (str): The column name to plot.
    title (str): Title of the plot.
    ylabel (str): Label for the y-axis.
    """
    # Ensure 'Datetime_start' is in datetime format
    # Use .loc to avoid SettingWithCopyWarning
    df = df.copy()  # Create an independent copy of the DataFrame
    df.loc[:, 'Datetime_start'] = pd.to_datetime(df['Datetime_start'])
    
    plt.figure(figsize=(12, 6))
    plt.plot(df['Datetime_start'], df[column], label=column, color='blue')
    
    # Format the x-axis for better readability
    ax = plt.gca()
    ax.xaxis.set_major_formatter(mdates.DateFormatter('%Y-%m-%d %H:%M'))  # Format as 'Year-Month-Day Hour:Minute'
    ax.xaxis.set_major_locator(mdates.AutoDateLocator())  # Automatically choose tick locations
    plt.xticks(rotation=45, fontsize=10)  # Rotate labels to avoid overlap
    
    # Add labels, title, and legend
    plt.title(title, fontsize=14)
    plt.xlabel('Datetime', fontsize=12)
    plt.ylabel(ylabel, fontsize=12)
    plt.legend(fontsize=12)
    plt.tight_layout()  # Adjust layout to prevent clipping
    plt.grid(True, linestyle='--', alpha=0.6)  # Add grid lines for better readability
    plt.show()

# Function to fetch raw data from the database
def fetch_raw_data(cursor, entity_name, start_date, end_date):
    query = """
    SELECT d.last_updated_ts, d.state
    FROM states d
    JOIN states_meta a ON d.metadata_id = a.metadata_id
    WHERE a.entity_id = %s COLLATE utf8mb4_unicode_ci
    AND last_updated_ts >= %s AND last_updated_ts < %s
    """
    cursor.execute(query, (entity_name, start_date.timestamp(), end_date.timestamp()))
    rows = cursor.fetchall()
    
    raw_data = pd.DataFrame(rows)
    
    if len(raw_data) > 0:
        # Remove non-numeric values from 'state' column
        raw_data['state'] = pd.to_numeric(raw_data['state'], errors='coerce')
        raw_data = raw_data.dropna(subset=['state'])
        # Convert 'last_updated_ts' to datetime
        raw_data['timestamp'] = pd.to_datetime(raw_data['last_updated_ts'], unit='s')
    
    return raw_data
